# Comandos para fazer deploy

# 1. Navegar para o diretório
cd C:\Users\Wesley\Desktop\instagram-scraper

# 2. Inicializar git
git init

# 3. Adicionar todos os arquivos
git add .

# 4. Fazer commit
git commit -m "Instagram scraper service with Playwright"

# 5. Adicionar remote
git remote add origin https://github.com/teste00373i-arch/instagram-scrape.git

# 6. Renomear branch para main
git branch -M main

# 7. Push para GitHub
git push -u origin main
